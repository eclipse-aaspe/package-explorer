// For Win-native checkable menu items
#if WINDOWS
using WinUiCtrl = Microsoft.UI.Xaml.Controls;
using Microsoft.Maui.Handlers;
#endif

// For macOS-native checkable menu items
#if MACCATALYST
using AppKit;
using Microsoft.Maui.Handlers;
#endif




        protected bool showGrid = true;

        protected void OnMenuItemHandlerChangedForCheckableMenuItems(object? sender, EventArgs e)
        {
            if (sender is not MenuFlyoutItem item)
                return;

#if WINDOWS
            if (item.Handler?.PlatformView is WinUiCtrl.MenuFlyoutItemBase baseItem)
            {
                // Windows native logic
                var toggle = new WinUiCtrl.ToggleMenuFlyoutItem
                {
                    Text = "Show Grid",
                    IsChecked = showGrid
                };

                toggle.Click += (_, __) =>
                {
                    showGrid = !toggle.IsChecked;
                    toggle.IsChecked = showGrid;
                };

                // Replace native item
                var x = baseItem.Parent;
                if (x is WinUiCtrl.MenuBarItem y)
                {
                    y.Items.Remove(baseItem);
                    y.Items.Add(toggle);
                }
            }
#endif

#if MACCATALYST
    if (item.Handler?.PlatformView is NSMenuItem native)
    {
        // macOS native logic
    }
#endif
        }
